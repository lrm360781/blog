前言
===
在开发后台时，为保证后台数据安全，我们要防止用户非法登录；且为增加用户体验，系统通过验证登录状态，避免用户重复登录。

实现步骤
===
先构建后台基础模板Base.php,后台所有controller继承该模板，代码如下：
```php
<?php
namespace app\index\controller;

use think\Controller;
use think\Session;
USE think\Request;

class Base extends Controller
{
    protected function _initialize()
    {
        parent::_initialize();
        define('USER_ID', Session::has('user_id') ? Session::get('user_id'):null);
        
        //实例化request
        $request=Request::instance();
        $con=$request->controller(); //获取控制器名
        $action=$request->action();  //获取方法名称
        $name=$con.'/'.$action;  
        //未检测session允许访问登录页面以及验证登录方法 
        //注：控制器大写/方法全小写
        $notChec=array('User/login','User/checklogin');
        //var_dump($name);die;

        //未登录后台时只允许进入user/login
        if(!in_array($name,$notChec)){
            $this->isLogin();
        }
    }
    
    //拦截未登录用户
    protected function isLogin()
    {
        if (is_null(USER_ID)) {
            $this -> error('用户未登陆,无权访问',url('user/login'));
        }
    }

    //防止用户重复登陆,放在登陆操作前面:user/login
    protected function alreadyLogin(){
        if (USER_ID) {
            $this -> error('用户已经登陆,请勿重复登陆',url('index/index'));
        }
    }
}
```
避免用户登录之后，在同一平台二次登录
```php
<?php
namespace app\index\controller;

class User extends Base
{
    //登陆界面
    public function login()
    {
        $this -> alreadyLogin();
        return $this -> view -> fetch();
    }
}
```
